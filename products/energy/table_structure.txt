#权限表
CREATE TABLE `authority_permission` (
  `permission_id` varchar(64) NOT NULL COMMENT '权限标示符',
  `permission_name` varchar(64) DEFAULT NULL COMMENT '权限名称',
  `permission_alias` varchar(2048) DEFAULT NULL,
  `permission_origin_id` varchar(64) DEFAULT NULL COMMENT '母权限标示符',
  PRIMARY KEY (`permission_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC;

CREATE TABLE `authority_role` (
  `role_id` varchar(64) NOT NULL COMMENT '角色标示符',
  `role_name` varchar(64) DEFAULT NULL COMMENT '角色名称',
  `role_type` char(1) DEFAULT NULL COMMENT '角色类型 是否超级管理员 1：是 2：否',
  `role_description` varchar(256) DEFAULT NULL COMMENT '角色描述',
  `organization_id` varchar(64) DEFAULT NULL COMMENT '组织标识符',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`role_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC COMMENT='用户';

CREATE TABLE `authority_role_permission_associate` (
  `role_id` varchar(64) NOT NULL COMMENT '角色标示符',
  `permission_id` varchar(64) DEFAULT NULL COMMENT '权限标示符',
  UNIQUE KEY `UNIQUE` (`role_id`,`permission_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC;

CREATE TABLE `authority_user` (
  `user_id` varchar(64) NOT NULL COMMENT '用户标示符',
  `user_account` varchar(64) DEFAULT NULL COMMENT '用户账号 （同一组织内不可重复）',
  `user_password` varchar(64) DEFAULT NULL COMMENT '用户密码（Hash密码）',
  `open_id` varchar(64) DEFAULT NULL COMMENT '微信用户标示',
  `user_sex` char(1) DEFAULT NULL COMMENT '用户性别 1：男 2：女',
  `user_phone` varchar(32) DEFAULT NULL COMMENT '用户手机号',
  `user_real_name` varchar(32) DEFAULT NULL COMMENT '用户名称',
  `user_icon` varchar(255) DEFAULT NULL COMMENT '用户头像路径',
  `user_idCard` varchar(32) DEFAULT NULL COMMENT '用户身份证号',
  `remark` varchar(255) DEFAULT NULL COMMENT '备注信息',
  `organization_id` varchar(64) DEFAULT NULL COMMENT '组织标识符',
  `user_status` char(1) DEFAULT '1' COMMENT '用户状态 1开启 2关闭',
  `create_time` datetime DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `last_modify_time` timestamp NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`user_id`) USING BTREE,
  UNIQUE KEY `UNIQUE-open_id` (`open_id`) USING BTREE,
  UNIQUE KEY `UNIQUE-user_phone` (`user_phone`) USING BTREE,
  UNIQUE KEY `UNIQUE-userIdCard` (`user_idCard`) USING BTREE,
  UNIQUE KEY `UNIQUE-account_org` (`user_account`,`organization_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC;
